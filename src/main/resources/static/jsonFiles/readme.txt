- Format: 7zipped json
 - Files:

     - Id
     - DocTagId
     - IsHelloWorldTopic

     - CreationDate
     - ViewCount
     - LastEditDate
     - ContributorCount

     - HelloWorldVersionsHtml
     - VersionsJson
     - ExampleCount
     - ExampleScore

     - SyntaxMarkdown

     - Id
     - DocTopicId

     - Id
